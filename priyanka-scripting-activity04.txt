#i/bin/bash

echo "enter password"
read password

length=${#password}

echo $length

if  [[ $length -lt 8 ]]
then	
	echo "your password should be more than 8 character"

	exit 1
fi


if [[ !($password =~ [0-9]) ]];
then
	echo "password should contain number"
	exit 1
fi

if [[ !($password =~ [a-z] && $password=~[A-Z]) ]];
then
        echo "password should contain uppercase and lowercase"

	exit 1
else 
   echo "your password is created"
fi


exit 0
